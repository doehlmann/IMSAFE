114 static int32_t connect_to_sensor(void)
115 {
116  char com_port_list[512];
117  char* current_port;
118 
119  /* retrieve a list of all available com ports */
120  com_port_list[0] = 0;
121  com_get_port_list(com_port_list, sizeof(com_port_list));
122 
123  /* iterate over the port list */
124  current_port = com_port_list;
125  while (current_port[0])
126  {
127  int32_t protocol_handle;
128 
129  /*
130  * find end of current port name in port list
131  * (port name is delimited by other ';' or '\0'
132  */
133  char* next_port = current_port;
134  while (next_port[0])
135  {
136  if (next_port[0] == ';')
137  {
138  /* replace ';' by '\0' to indicate end of current port name */
139  *next_port = 0;
140 
141  /* after a ';' another port name follows */
142  ++next_port;
143  break;
144  }
145  ++next_port;
146  }
147 
148  /* try to connect to current COM port */
149  printf("Trying to open COM port: %s ... ", current_port);
150  protocol_handle = protocol_connect(current_port);
151 
152  if (protocol_handle >= 0)
153  {
154  printf("OK.\n");
155  return protocol_handle;
156  }
157  else
158  {
159  printf("Failed.\n");
160  }
161 
162  /* prepare for next iteration */
163  current_port = next_port;
164  }
165 
166  return -1;
167 }
