177 static int32_t parse_frame_info(int32_t protocol_handle, uint8_t endpoint,
178  const uint8_t* payload,
179  uint16_t payload_size)
180 {
181  get_raw_data_cnf_t frame_info;
182  uint32_t total_samples;
183  uint32_t expected_message_size;
184 
185  uint8_t header_size = 18;
186 
187  /* read header info from payload */
188  frame_info.frame_number =
189  protocol_read_payload_uint32(payload, 1);
190  frame_info.num_chirps = protocol_read_payload_uint32(payload, 5);
191  frame_info.num_rx_antennas =
192  protocol_read_payload_uint8(payload, 9);
193  frame_info.num_samples_per_chirp =
194  protocol_read_payload_uint32(payload, 10);
195  frame_info.rx_mask = protocol_read_payload_uint8(payload, 14);
196  frame_info.data_format =
197  (Rx_Data_Format_t)protocol_read_payload_uint8(payload, 15);
198  frame_info.adc_resolution =
199  protocol_read_payload_uint8(payload, 16);
200  frame_info.interleaved_rx =
201  protocol_read_payload_uint8(payload, 17);
202 
203  /* calculate expected message size */
204  total_samples = frame_info.num_chirps *
205  frame_info.num_samples_per_chirp *
206  frame_info.num_rx_antennas *
207  (frame_info.data_format == EP_RADAR_BASE_RX_DATA_REAL ? 1 : 2);
208 
209  expected_message_size = total_samples * frame_info.adc_resolution;
210  expected_message_size = header_size +
211  (expected_message_size >> 3) +
212  ((expected_message_size & 0x07) ? 1 : 0);
213 
214  if (payload_size == expected_message_size)
215  {
216  uint32_t sample_bit_mask = (1 << frame_info.adc_resolution) - 1;
217  const float norm_factor = 1.f / (float)sample_bit_mask;
218  float *sample_data = (float *)malloc(total_samples *
219  sizeof(float));
220  float *write_ptr = sample_data;
221 
222  uint16_t read_idx = header_size;
223  uint16_t read_bit_position = 0;
224 
225  while (total_samples)
226  {
227  uint32_t current_sample =
228  protocol_read_payload_uint16(payload, read_idx) >>
229  read_bit_position;
230  *write_ptr++ = (current_sample & sample_bit_mask) *
231  norm_factor;
232 
233  read_bit_position += frame_info.adc_resolution;
234  read_idx += read_bit_position >> 3;
235  read_bit_position &= 0x07;
236 
237  --total_samples;
238  }
239 
240  /* send frame info to callback */
241  frame_info.sample_data = sample_data;
242  get_raw_data_cnf_cb(get_raw_data_cnf_ctx, protocol_handle,
243  endpoint, &frame_info);
244  free(sample_data);
245  }
246 
247  return 1;
248 }
