167 /*
168  * The max str. len that is transmitted is only 255 bytes (because a uint8_t is used for the size
169  *
170  */
171 static uint16_t protocol_read_helper_str_list(const uint8_t *payload, const size_t offset, size_t *p_elements, char ***ptr )
172 {
173  uint16_t increment = 0;
174  uint8_t elements = protocol_read_payload_uint8( &payload[0], (uint16_t)offset );
175  increment += 1;
176  if( p_elements != NULL)
177  *p_elements = (size_t) elements;
178 
179  if( ptr == NULL)
180  return 0;
181  char **pData = (char **) malloc(sizeof(char**) * elements );
182  for( int i = 0; i < elements; i++)
183  {
184  uint8_t len = protocol_read_payload_uint8( &payload[0], (uint16_t)offset + increment );
185  increment += 1;
186  char* pTemp = (char *) malloc(sizeof(char) * ( len + 1 ) );
187  memcpy(pTemp, &payload[offset + increment], len);
188  *(pTemp + len) = 0; // zero terminate the string!
189  pData[i] = pTemp;
190  increment += len;
191  }
192  *ptr = pData;
193  return increment;
194 }

203 static int32_t parse_para_values_cnf(int32_t protocol_handle, uint8_t endpoint,
204  const uint8_t* payload, uint16_t payload_size)
205 {
206  uint8_t payloadId = protocol_read_payload_uint8(payload, 0);
207 
208  if ((payloadId == MSG_CONSUMPTION_DEF_CNF) &&
209  (payload_size > 1))
210  {
211  if (consumption_def_cnf_cb)
212  {
213  // TODO: move this out into a seperate function!
214  consumption_def_host_t consumption_def;
215  uint16_t offset = 1; // skip message id
216 
217  uint16_t size = protocol_read_payload_uint16(&payload[0], offset);
218  offset += 2;
219 
220  offset += protocol_read_helper_str_list( &payload[0], offset, &consumption_def.consumption_label_elements, &consumption_def.pp_consumption_labels);
221 
222  // check if offset and size matches!!
223  if( size != offset )
224  return EP_RADAR_ERR_MSG_SIZE_CORRUPT;
225 
226  /* send frame format to callback */
227  consumption_def_cnf_cb(consumption_def_cnf_ctx, protocol_handle,
228  endpoint, &consumption_def);
229 
230  // free-up memory!!
231  for( size_t i = 0; i < consumption_def.consumption_label_elements; i++)
232  {
233  free(consumption_def.pp_consumption_labels[i]);
234  }
235  free(consumption_def.pp_consumption_labels);
236 
237 
238  }
239  return 1;
240  }
241  else if ((payloadId == MSG_CONSUMPTION_CNF) &&
242  (payload_size > 1))
243  {
244  if (consumption_cnf_cb)
245  {
246 
247  consumption_cnf_t consumption;
248 
249  uint16_t offset = 1;
250  size_t elements = 0;
251  uint16_t size = protocol_read_payload_uint16(&payload[0], offset);
252  offset += sizeof(uint16_t);
253 
254  elements = (size_t) protocol_read_payload_uint8(&payload[0], offset);
255  offset += sizeof(uint8_t);
256 
257  consumption_t *pp_temp = (consumption_t *) malloc( elements *sizeof(consumption_t) );
258 
259  for(uint32_t i = 0; i < elements; i++)
260  {
261  consumption_t *pData = (pp_temp+i);
262  pData->index = protocol_read_payload_uint8(&payload[0], offset);
263  offset += 1;
264  uint32_t temp;
265  temp = protocol_read_payload_uint32(&payload[0], offset);
266  memcpy(&pData->value, &temp, sizeof(float));
267  offset += sizeof(uint32_t);
268  size_t len = (size_t) protocol_read_payload_uint8(&payload[0], offset);
269  offset += 1;
270 
271  pData->p_unit = (char *)malloc(len + 1);
272  memcpy(pData->p_unit, &payload[offset], len);
273  pData->p_unit[len] = 0;
274 
275  offset += (uint16_t)len;
276 
277  }
278  consumption.elements = elements;
279  consumption.p_consumption = pp_temp;
280 
281 
282  if (size != offset)
283  return EP_RADAR_ERR_MSG_SIZE_CORRUPT;
284 
285  /* send result to callback */
286  consumption_cnf_cb(consumption_cnf_ctx, protocol_handle,
287  endpoint, &consumption);
288 
289  // free the memory again:
290  for( uint32_t i=0; i< elements; i++)
291  {
292  consumption_t *pData = consumption.p_consumption+i;
293  free(pData->p_unit);
294  }
295  free(consumption.p_consumption);
296 
297  }
298  return 1;
299  }
300  else if ((payloadId == MSG_BOARD_INFO_CNF) &&
301  (payload_size > 1))
302  {
303  if (board_info_cnf_cb)
304  {
305  board_info_cnf_t board_info;
306  uint16_t offset = 1;
307  board_info.major_version = protocol_read_payload_uint8(&payload[0], offset);
308  offset += 1;
309  board_info.minor_version = protocol_read_payload_uint8(&payload[0], offset);
310  offset += 1;
311 
312  uint8_t len = protocol_read_payload_uint8(&payload[0], offset);
313  offset += 1;
314  board_info.rf_shield_type_id = (char *)malloc(len+1);
315  memcpy(board_info.rf_shield_type_id, &payload[offset], len);
316  offset += len;
317  board_info.rf_shield_type_id[len] = 0;
318 
319  len = protocol_read_payload_uint8(&payload[0], offset);
320  offset += 1;
321  board_info.description = (char *)malloc(len+1);
322  memcpy(board_info.description, &payload[offset], len);
323  board_info.description[len] = 0;
324  offset += len;
325 
326  /* send result to callback */
327  board_info_cnf_cb(board_info_cnf_ctx, protocol_handle,
328  endpoint, &board_info);
329 
330  free( board_info.rf_shield_type_id);
331  free( board_info.description );
332  }
333 
334  }
335  else if ((payloadId == MSG_PARA_STORE_CNF) &&
336  (payload_size == 1))
337  {
338  if (para_store_cnf_cb)
339  {
340  para_store_cnf_t data;
341  /* send result to callback */
342  para_store_cnf_cb(para_store_cnf_ctx, protocol_handle, endpoint, &data);
343  }
344  }
345 
346  else if ((payloadId == MSG_PARA_FACTORY_RESET_CNF) &&
347  (payload_size == 1))
348  {
349  if (para_factory_reset_cnf_cb)
350  {
351  para_factory_reset_t data;
352  /* send result to callback */
353  para_factory_reset_cnf_cb(para_factory_reset_cnf_ctx, protocol_handle, endpoint, &data);
354  }
355  }
356 
357  else
358  {
359  printf("tjba: message id not found\n");
360  }
361 
362  return 0;
363 }

481 /*
482  * if repeat_count is 0xFFFF.FFFF the results will be continuously provided, unless stopped (with repeat_count set to 0)
483  * if repeat_count is 1, only a single result is returned, and the next result needs to be retriggerd
484  * in case you like to avoid extra latency, by the round trip request, set the result cnt to 0xfffffffff, and a continuous stream of results will be received.
485  */

487 int32_t ep_radar_base_board_consumption_req(int32_t protocol_handle, uint8_t endpoint, uint32_t repeat_count)
488 {
489  /* build message to send */

491  uint16_t offset = 0;
492 
493  offset += protocol_write_payload_uint8(cmd_message, offset, MSG_CONSUMPTION_REQ);
494  offset += protocol_write_payload_uint32(cmd_message, offset, repeat_count);
495 
496  /* send message and process received response */
497  return protocol_send_and_receive(protocol_handle, endpoint,
498  &ep_RadarBaseBoard_definition,
499  cmd_message, offset);
500 }
