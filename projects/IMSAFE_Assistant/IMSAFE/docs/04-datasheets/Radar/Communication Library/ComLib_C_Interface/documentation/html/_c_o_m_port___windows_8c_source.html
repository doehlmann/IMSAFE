104 static bool com_find(const char* vid, const char* pid,
105  void(*callback)(const char* com_port, void* context),
106  void* context)
107 {
108  if (!is_valid_vid_pid(vid, pid))
109  return false;
110 
111  // plug and play id: USB\VID_XXXX&PID_XXXX where XXXX are hexadecimal characters
112  char pnpid[32];
113  snprintf(pnpid, sizeof(pnpid), "USB\\VID_%c%c%c%c&PID_%c%c%c%c",
114  toupper(vid[0]), toupper(vid[1]), toupper(vid[2]), toupper(vid[3]),
115  toupper(pid[0]), toupper(pid[1]), toupper(pid[2]), toupper(pid[3]));
116 
117  // SetupDiGetClassDevs returns a handle to a device information set that
118  // contains requested device information elements for a local computer
119  HDEVINFO DeviceInfoSet = SetupDiGetClassDevs(NULL, TEXT("USB"), NULL,
120  DIGCF_ALLCLASSES | DIGCF_PRESENT);
121  if (DeviceInfoSet == INVALID_HANDLE_VALUE)
122  return false;
123 
124  DWORD DeviceIndex = 0;
125  SP_DEVINFO_DATA DeviceInfoData = {0};
126  DeviceInfoData.cbSize = sizeof(SP_DEVINFO_DATA);
127 
128  // SetupDiEnumDeviceInfo returns a SP_DEVINFO_DATA structure that specifies
129  // a device information element in a device information set
130  while (SetupDiEnumDeviceInfo(DeviceInfoSet, DeviceIndex++, &DeviceInfoData))
131  {
132  BYTE buffer[256] = {0};
133  DEVPROPTYPE ulPropertyType;
134  DWORD dwSize1;
135 
136  // get specified plug and play device property
137  if (SetupDiGetDeviceRegistryPropertyA(DeviceInfoSet, &DeviceInfoData,
138  SPDRP_HARDWAREID, &ulPropertyType,
139  buffer, sizeof(buffer), &dwSize1))
140  {
141  // check if this is the device we are looking for
142  if (strncmp((char*)buffer, pnpid, strlen(pnpid)) != 0)
143  continue;
144 
145  // open registry
146  HKEY hDeviceRegistryKey = SetupDiOpenDevRegKey(DeviceInfoSet,
147  &DeviceInfoData,
148  DICS_FLAG_GLOBAL, 0,
149  DIREG_DEV, KEY_READ);
150  if (hDeviceRegistryKey == INVALID_HANDLE_VALUE)
151  continue;
152 
153  // query
154  char port_name[32] = {0};
155  DWORD dwSize2 = sizeof(port_name);
156  DWORD dwType = REG_SZ;
157  LSTATUS ret = RegQueryValueExA(hDeviceRegistryKey, "PortName",
158  NULL, &dwType, (LPBYTE)port_name, &dwSize2);
159 
160  // close registry key
161  RegCloseKey(hDeviceRegistryKey);
162 
163  if (ret == ERROR_SUCCESS && (dwType == REG_SZ))
164  {
165  // check if the usb device corresponds to a COM port
166  if (strncmp(port_name, "COM", 3) == 0)
167  callback(port_name, context);
168  }
169  }
170  }
171 
172  SetupDiDestroyDeviceInfoList(DeviceInfoSet);
173 
174  return true;
175 }

177 static void callback(const char* com_port, void* context)
178 {
179  Port_List_t* ports = (Port_List_t*)context;
180  const size_t len = strlen(com_port);
181 
182  if (ports->ports_available == 0 && ports->buffer_available >= len)
183  {
184  memcpy(ports->port_list, com_port, len);
185  ports->port_list += len;
186  ports->buffer_available -= len;
187  }
188  else if (ports->ports_available > 0 && ports->buffer_available >= (len + 1))
189  {
190  *ports->port_list++ = ';';
191  memcpy(ports->port_list, com_port, len);
192  ports->port_list += len;
193  ports->buffer_available -= len + 1;
194  }
195 
196  ports->ports_available++;
197 }

220 com_t* com_open(const char* port_name)
221 {
222  COMMTIMEOUTS timeouts;
223  DCB com_configuration = {0};
224 
225  com_t* com_port = malloc(sizeof(com_t));
226  if (com_port == NULL)
227  return NULL;
228 
229  /*
230  * use full path because Windows won't find "COM10" and higher
231  * (see knowledge base http://support.microsoft.com/kb/115831/en-us)
232  *
233  * To work with both Multi-Byte-Charset and Unicode Charset convert
234  * port_name to TCHAR, because that type is needed by Win32 API.
235  * I did not find a suitable out-of-the-box function for this conversion
236  * so it's done explicitly here.
237  */
238  #define T_BUFFER_LENGTH 32
239  TCHAR full_port_name[T_BUFFER_LENGTH] = TEXT("\\\\.\\");
240  unsigned idx = 0;
241  while ((port_name[idx] != 0) && (idx + 5 < T_BUFFER_LENGTH))
242  {
243  full_port_name[4 + idx] = port_name[idx];
244  ++idx;
245  }
246  full_port_name[4 + idx] = 0;
247 
248  /* open the COM port */
249  /* ----------------- */
250  com_port->handle = CreateFile(full_port_name, /* name of the COM port */
251  /*lint -e620 we don't really have a chance to change a MSYS/MinGW header file ... */
252  GENERIC_READ | /* access type */
253  /*lint -e620 */
254  GENERIC_WRITE,
255  0, /* shared mode */
256  NULL, /* security attributes */
257  OPEN_EXISTING, /* creation disposition */
258  0, /* flags and attributes */
259  0); /* template file */
260 
261  /* if COM port could not be opened, return negative Windows error code */
262  if (com_port->handle == INVALID_HANDLE_VALUE)
263  {
264  return NULL;
265  }
266 
267  /* set timeouts */
268  timeouts.ReadIntervalTimeout = 0;
269  timeouts.ReadTotalTimeoutMultiplier = 0;
270  timeouts.ReadTotalTimeoutConstant = 1000;
271  timeouts.WriteTotalTimeoutConstant = 100;
272  timeouts.WriteTotalTimeoutMultiplier = 1;
273  (void)SetCommTimeouts(com_port->handle, &timeouts);
274 
275  /* configure COM Port (even though it's virtual) */
276  com_configuration.DCBlength = sizeof(DCB);
277  (void)GetCommState (com_port->handle, &com_configuration);
278  com_configuration.BaudRate = 115200;
279  com_configuration.ByteSize = 8;
280  com_configuration.StopBits = ONESTOPBIT;
281  (void)SetCommState (com_port->handle, &com_configuration);
282 
283  return com_port;
284 }
